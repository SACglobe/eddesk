My Database:
Table: schools
Columns:
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
name TEXT NOT NULL,
slug TEXT UNIQUE,                 -- for internal reference
custom_domain TEXT UNIQUE,        -- abcschool.com ( eddesk.in for                        demo )
logo_url TEXT,
email TEXT,
phone TEXT,
address TEXT,
city TEXT,
state TEXT,
country TEXT,
postal_code TEXT,
plan_type TEXT CHECK (plan_type IN ('basic','pro','premium')),
subscription_status TEXT,
is_active BOOLEAN DEFAULT true,
created_at TIMESTAMP DEFAULT now(),
updated_at TIMESTAMP DEFAULT now()
Template_id - template_classic / template_modern / template_premium
expiration_date
Paymentgateway_url


table: homepage_sections
Columns:
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
school_id UUID REFERENCES schools(id) ON DELETE CASCADE,
section_key TEXT,  -- hero, stats, achievements, principal, faculty, sports, etc.
is_enabled BOOLEAN DEFAULT true,
display_order INT,
settings JSONB,    -- section-specific config
created_at TIMESTAMP DEFAULT now()

table: hero_media
Columns:
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
school_id UUID REFERENCES schools(id) ON DELETE CASCADE,
media_type TEXT CHECK (media_type IN ('image','video')),
media_url TEXT,
headline TEXT,
subheadline TEXT,
primary_button_text TEXT,
primary_button_url TEXT,
secondary_button_text TEXT,
secondary_button_url TEXT,
display_order INT,
is_active BOOLEAN DEFAULT true

table: ANNOUNCEMENTS
Columns:
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

title TEXT,
message TEXT,
priority INT DEFAULT 1,

is_active BOOLEAN DEFAULT true
